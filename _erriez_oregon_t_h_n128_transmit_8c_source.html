191 void OregonTHN128_TxRawData(uint32_t rawData)
192 {
193  /* Check RF transmit pin initialized */
194  if ((_rfTxPort < 0) && (_rfTxBit < 0)) {
195  return;
196  }
197 
198  /* First transmit */
199  txPreamble();
200  txSync();
201  txData(rawData);
202  txDisable();
203 
204  /* Wait 100ms between frames */
205  if (delay100ms != NULL) {
206  /* Call 100ms user delay (weak symbol) */
207  delay100ms();
208  } else {
209  /* Use optimized AVR delay */
210  _delay_ms(T_SPACE_FRAMES_MS);
211  }
212 
213  /* Second transmit */
214  txPreamble();
215  txSync();
216  txData(rawData);
217  txDisable();
218 }
